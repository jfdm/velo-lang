data AtIndex : (ty : kind) -> (ctxt : SnocList kind) ->
               (idx : Nat) -> Type where
  Here : AtIndex ty (ctxt :< ty) 0
  There : AtIndex ty ctxt idx -> AtIndex ty (ctxt :< _) (1 + idx)
